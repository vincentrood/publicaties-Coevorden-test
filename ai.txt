export default {
  async fetch(request, env) {
    // Afhandeling van CORS (zodat jouw website toestemming heeft om deze worker aan te roepen)
    const corsHeaders = {
      "Access-Control-Allow-Origin": "*", // Voor extra veiligheid later vervangen door je echte domein
      "Access-Control-Allow-Methods": "POST, OPTIONS",
      "Access-Control-Allow-Headers": "Content-Type",
    };

    // Reageer op 'preflight' verzoeken van de browser
    if (request.method === "OPTIONS") {
      return new Response(null, { headers: corsHeaders });
    }

    if (request.method !== "POST") {
      return new Response("Stuur een POST verzoek", { status: 405 });
    }

    try {
      const { context, question } = await request.json();

      const response = await fetch("https://api.openai.com/v1/chat/completions", {
        method: "POST",
        headers: {
          "Authorization": `Bearer ${env.OPENAI_API_KEY}`,
          "Content-Type": "application/json"
        },
        body: JSON.stringify({
          model: "gpt-4o-mini", // Of gpt-3.5-turbo, maar 4o-mini is slim en goedkoop
          messages: [
            { 
              role: "system", 
              content: "Je bent een behulpzame assistent voor de gemeente. Beantwoord vragen uitsluitend op basis van de meegeleverde tekst van het document. Als het antwoord niet in de tekst staat, geef dat dan aan." 
            },
            { 
              role: "user", 
              content: `Document tekst:\n${context}\n\nGebruikersvraag: ${question}` 
            }
          ],
          max_tokens: 1000
        })
      });

      const data = await response.json();
      const aiAnswer = data.choices[0].message.content;

      return new Response(JSON.stringify({ answer: aiAnswer }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" }
      });

    } catch (err) {
      return new Response(JSON.stringify({ error: err.message }), {
        status: 500,
        headers: corsHeaders
      });
    }
  }
};
